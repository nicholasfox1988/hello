  <button disabled>同意(5)</button>

  <script>
    let btn1 = document.querySelector('button')
    let i = 5
    let n = setInterval(() => {
      i--
      btn1.innerHTML = `同意(${i})`
      if (i === 0) {
        clearInterval(n)
        btn1.disabled = false
        btn1.innerHTML = `同意.`
      }
    }, 1000)
  </script>
